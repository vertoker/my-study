#include <iostream> 
using namespace std;
int main()
{

    int j,k=8;
    for (int i=1; i<=5;i++)
    {
        for (j=1;j<=k;j++)
        {
            cout<<i<<" ";
        }
        k--;
        cout<<"\n";
    }
    return 0;
}
