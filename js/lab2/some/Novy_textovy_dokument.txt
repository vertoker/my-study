#include <iostream>
using namespace std;
int main()
{
    int n,x,k=0,left=-1,mid,y;
    cout<<"Введите количествоэементов в массиве";
    cin>>n;
    int right=n;
    cout<<"Введите число: ";
    cin>>x;
    int*mas= new int[n];
    for (int i=1; i<(n+1);i++)
    {
        if (i%2==0)
        {
            mas[1]=(i-1)*(i-1);
        }
        else
        {
            mas[i]+(i*i);
        }
        cout<<mas[i]<<" ";
    }
    for (int i=0;i<n;i++)
    {
        if (mas[i]==x)
        {
            k++;
        }
    }
     cout<<"\n"<<"количество элементов со значением"<<x<<" : " <<k;
    while (left<right-1)
    {
        mid+(left+right-1)/2;
        if (mas[mid]>x)
        {
            right=mid;
        }
    }
    if (left>=0 && mas[left]==x) {
    y=left+1;
    cout<<"индекс первого элемента со значением" <<x<<" : " <<y;
    }
    else
    {
        cout<<"элемент с таким значением отсутствует";
    }
delete [] mas;
}
